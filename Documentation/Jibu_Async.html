Remarks

This class extends the abstract Task class, but also exposes the WaitFor and Start methods. When creating an Async task, you must implement the abstract Run method. Start immediately schedules the Async for execution. WaitFor waits for the Task to finish execution. Any uncaught exception thrown by the task will be rethrown by WaitFor, encapsulated in a CancelException. If the Async is cancelled, then WaitFor throws a CancelException. 

For information about the Jibu scheduler and examples demonstrating the mailbox system, please refer to the Task class.

Async Example
Copy Code
using System;
using Jibu;

namespace AsyncExample
{
    class ExampleTask : Async
    {
        private int iterations;

        public ExampleTask(int iterations)
        {
            this.iterations = iterations;
        }

        public override void Run()
        {
            for (int i = 0; i < iterations; i++)
            {
                Console.WriteLine("iteration: " + i);
                Timer.SleepFor(500);
            }
        }
    }

    class AsyncExample
    {
        static void Main(string[] args)
        {
            // Create a Task that immediately schedules the task 
            // for execution.
            Async t = new ExampleTask(10).Start();

            // Do some work here.... we just sleep
            Timer.SleepFor(2000);

            // wait for the task to finish
            Console.WriteLine("Main thread waiting...");
            t.WaitFor();

            // Create a Task with a task that will throw an exception
            Async t2 = new DelegateAsync(delegate 
            { 
                throw new Exception("Goodbye World!"); 
            }).Start();

            // Do some work here.... again we just sleep
            Timer.SleepFor(2000);

            try
            {
                // this should throw a CancelException
                // wrapping the exception thrown by
                // our task.
                t2.WaitFor();
            }
            catch (CancelException e)
            {
                // We check the InnerException property.                
                if(e.InnerException != null)
                    Console.WriteLine("Success! We caught an exception: " + e.InnerException.Message);
            }

        }
    }
}
