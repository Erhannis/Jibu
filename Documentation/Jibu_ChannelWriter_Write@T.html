Remarks

The Write method writes data to the Channel. If the ChannelWriter is the writing end of a buffered Channel, Write will return once it has written data to the buffer. If the ChannelWriter is a writing end of an unbuffered Channel, Write returns when, and only when, the data is read by another task.

Unbuffered Channel Example
Copy Code
using System;
using Jibu;

namespace ChannelExample
{
    // Demonstrates the unbuffered Channel
    class UnbufferedChannelExample
    {
        static void Main(string[] args)
        {
            // Create unbuffered Channel that 
            // can transfer integer values
            Channel<int> chan = new Channel<int>();

            // It is good practice - but not required -
            // to separate the reading and writing ends
            // of the Channel
            ChannelReader<int> readEnd = chan.ChannelReader;
            ChannelWriter<int> writeEnd = chan.ChannelWriter;


            // This tasks writes an integer each second
            Async writer = new DelegateAsync(delegate
            {
                int i = 0;

                while (true)
                {
                    // Since this channel is unbuffered 
                    // this call hangs until the 
                    // reader has read the data.
                    writeEnd.Write(i++);
                    Timer.SleepFor(500);
                }
            }).Start();

            // This task reads an integer and 
            // prints it to the console
            Async reader = new DelegateAsync(delegate
            {
                while (true)
                {
                    // This call hangs until data 
                    // is written to the channel.
                    int i = readEnd.Read();
                    Console.WriteLine("Read " + i);
                }
            }).Start();


            Timer.SleepFor(5000);

            // When we poison the channel both 
            // tasks will receive a PoisonException
            // when they call read or write.
            chan.Poison();

            try
            {
                reader.WaitFor();
            }
            catch (CancelException ex)
            {
                if (ex.InnerException != null && ex.InnerException is PoisonException)
                    Console.WriteLine("The reader threw PoisonException");
            }

            try
            {
                writer.WaitFor();
            }
            catch (CancelException ex)
            {
                if (ex.InnerException != null && ex.InnerException is PoisonException)
                    Console.WriteLine("The writer threw PoisonException");
            }

        }
    }
}
Buffered Channel Example
Copy Code
using System;
using Jibu;

namespace BufferedChannelExample
{
    // Demonstrates the buffered Channel
    class BufferedChannelExample
    {
        static void Main(string[] args)
        {
            int bufferSize = 5;

            // Create buffered Channel that 
            // can transfer integer values
            Channel<int> chan = new Channel<int>(JibuBuffer.Fifo, bufferSize);

            // It is good practice - but not required -
            // to separate the reading and writing ends
            // of the Channel
            ChannelReader<int> readEnd = chan.ChannelReader;
            ChannelWriter<int> writeEnd = chan.ChannelWriter;

            // This tasks writes bufferSize integers 
            // each second, and hangs if the numbers 
            // aren't comsumed by a reader.
            Async writer = new DelegateAsync(delegate
            {
                int val = 0;
                while (true)
                {
                    for (int i = 0; i < bufferSize; i++)
                        writeEnd.Write(val++);
                }
            }).Start();

            // we read bufferSize elements from the buffer.
            for (int i = 0; i < bufferSize; i++)
            {
                Console.WriteLine("Read from Channel: {0}", readEnd.Read());
            }

            // we sleep for a while and let the writer 
            // write some data to the buffer.
            Timer.SleepFor(1000);

            // When we poison the channel the writer 
            // task will throw a PoisonException...            
            chan.Poison();

            try
            {                
                writer.WaitFor();
            }
            catch (CancelException ex)
            {
                if(ex.InnerException != null && ex.InnerException is PoisonException)
                    Console.WriteLine("Task threw PoisonException");

                
            }

            // ... but we will be able to read the 
            // remaining buffer elements
            for (int i = 0; i < bufferSize; i++)
            {
                Console.WriteLine("Read after poison: {0}", chan.Read());
            }

            
            try
            {
                // This will throw a PoisonException since 
                // the Channel is poisoned and there 
                // are no more elements in the buffer
                readEnd.Read();
            }
            catch (PoisonException)
            {
                Console.WriteLine("Channel is poisoned");
            }
        }
    }
}
