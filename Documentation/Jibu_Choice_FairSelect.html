Fair selection of alternatives

C++
public: int FairSelect();
C#
public int FairSelect();
Visual Basic
Public Function FairSelect() As Integer
Returns

An index of a ready alternative.

Remarks

FairSelect guaranties that a ready alternative will be selected within n calls to FairSelect, where n is the number of submitted alternatives. FairSelect is blocked until one of the alternatives are ready. 

 

FairSelectExample
Copy Code
using System;
using System.Collections.Generic;
using System.Text;
using Jibu;

namespace FairSelectExample
{
    class FairSelectExample
    {
        static void Main(string[] args)
        {
            Channel<Tuple<int, string>> chan1 = new Channel<Tuple<int, string>>(JibuBuffer.Infinite);
            Channel<int> chan2 = new Channel<int>(JibuBuffer.Infinite);

            Parallel.Run(new FairSelectClass(chan1.ChannelReader, chan2.ChannelReader), new Writer1(chan1.ChannelWriter), new Writer2(chan2.ChannelWriter));

        }
    }

    // This task reads 20 elements from two channels
    // using fair selection
    public class FairSelectClass : Async
    {
        ChannelReader<Tuple<int, string>> reader1;
        ChannelReader<int> reader2;

        public FairSelectClass(ChannelReader<Tuple<int, string>> reader1, ChannelReader<int> reader2)
        {
            this.reader1 = reader1;
            this.reader2 = reader2;
        }

        public override void Run()
        {
            // Create a Choice, that is listening
            // on the two read-ends.
            Choice c = new Choice(reader1, reader2);
            Tuple<int, string> data;
            for (int i = 0; i < 20; i++)
            {
                // Fair select  
                switch (c.FairSelect())
                {
                    case 0:
                        data = reader1.Read();
                        Console.WriteLine(data.First + " " + data.Second);
                        break;
                    case 1:
                        Console.WriteLine(reader2.Read());
                        break;
                }
            }
        }
    }

    // This task writes 10 integer/string pairs
    // to a channel
    public class Writer1 : Async
    {
        ChannelWriter<Tuple<int, string>> writer1;

        public Writer1(ChannelWriter<Tuple<int, string>> writer1)
        {
            this.writer1 = writer1;
        }

        public override void Run()
        {
            for (int i = 0; i < 10; i++)
                writer1.Write(new Tuple<int, string>(i, "Hello"));
        }
    }

    // This task writes 10 integers to a channel
    public class Writer2 : Async
    {
        ChannelWriter<int> writer2;

        public Writer2(ChannelWriter<int> writer2)
        {
            this.writer2 = writer2;
        }

        public override void Run()
        {
            for (int i = 0; i < 10; i++)
                writer2.Write(i);
        }
    }
