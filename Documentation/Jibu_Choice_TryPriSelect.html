Non-blocked prioritized selection of alternatives.

C++
public: int TryPriSelect();
C#
public int TryPriSelect();
Visual Basic
Public Function TryPriSelect() As Integer
Returns

An index of a ready alternative with the highest priority or -1 if no alternative is ready.

Remarks

Returns an index of a ready alternative. If multiple alternatives are ready, the one with the highest priority is selected. If no alternatives are ready, -1 is returned. 

 

TryPriSelectExample
Copy Code
using System;
using System.Collections.Generic;
using System.Text;
using Jibu;

namespace TryPriSelectExample
{
    class TryPriSelectExample
    {
        static void Main(string[] args)
        {
            Channel<int> intChan = new Channel<int>();
            Channel<string> stringChan = new Channel<string>();

            // Create a Task that writes 20 numbers to the intChan in 20 ms intervals
            new DelegateAsync(delegate {
                for (int i = 0; i < 20; i++)
                {
                    intChan.Write(i);
                    Timer.SleepFor(10);
                }

            }).Start();

            // Create a Task that reads 20 strings from  stringChan in 20 ms intervals
            new DelegateAsync(delegate
            {
                for (int i = 0; i < 20; i++)
                {
                    Console.WriteLine(stringChan.Read());
                    Timer.SleepFor(20);
                }
            }).Start();

            // Create the ChoiceTask that uses Choice to listen for events on the two channels
            Parallel.Run(new TryPriSelectClass(intChan.ChannelReader, stringChan.ChannelWriter));           
        
        }
    }

    class TryPriSelectClass : Async
    {
        private ChannelReader<int> input;
        private ChannelWriter<string> output;

        public TryPriSelectClass(ChannelReader<int> input, ChannelWriter<string> output)
        {
            this.input = input;
            this.output = output;
        }

        public override void Run()
        {
            // Create a choice object and connect a ChannelReader and a 
            // ChannelWriter.
            Choice choice = new Choice(input, output);
            int counter = 0;

            // Check if one of the alternatives are ready, if not move on
            // to do other stuff. TryPriSelect doesn't block if none of
            // the alternatives are ready - it just returns -1.
            while (counter < 40)
            {
                switch (choice.TryPriSelect())
                {
                    case 0:
                        Console.WriteLine(input.Read());
                        counter++;
                        break;
                    case 1:
                        output.Write("Some data " + counter);
                        counter++;
                        break;
                    case -1:
                        Console.WriteLine("Nothing ready");
                        Timer.SleepFor(10);
                        break;
                }
            }
        }
    }
}
