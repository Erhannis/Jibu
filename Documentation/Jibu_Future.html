An asynchronous Task that returns a result.

T
The type of the result

Remarks

This class extends the abstract Task class, but also exposes the Result and Start methods. When creating a Future task, you must implement the abstract Run method. Start immediately schedules the Future for execution. Result waits for the Task to finish execution and returns the result. Any uncaught exception thrown by the task will be rethrown by Result, encapsulated in a CancelException. If the Future is cancelled, then Result throws a CancelException. 

For information about the Jibu scheduler and examples demonstrating the mailbox system, please refer to the Task class.

Future Example
Copy Code
using System;
using Jibu;

namespace FutureExample
{
    // Simple task that can calculate Fibonacci numbers
    class FibonacciTask : Future<int>
    {
        int n;

        public FibonacciTask(int n)
        {
            this.n = n;    
        }

        private static int Fibonacci(int n)
        {
            if(n <= 2)
                return 1;
            else
                return Fibonacci(n - 2) + Fibonacci(n - 1);            
        }

        public override int Run()
        {
            return Fibonacci(n);
        }

    }

    class FutureExample
    {
        static void Main(string[] args)
        {
            int n = 40;

            // Start the asynchronous task
            Future<int> fibonacciTask = new FibonacciTask(n).Start();

            // You can do some work here....

            Console.WriteLine("Waiting for the Future to finish...");

            // ... and then get the result later
            Console.WriteLine("Fibonacci({0}) = {1} ", n, fibonacciTask.Result());


            // This task throws an exception...
            Future<int> exceptionTask = new DelegateFuture<int>(delegate 
            { 
                throw new Exception("Goodbye world!"); 
            }).Start();

            try
            {
                int res = exceptionTask.Result();
            }
            catch (CancelException e)
            {
                // ... but it isn't propagated out until 
                // we call the Result method.
                // Result throws the caught exception 
                // wrapped in a Jibu.CancelException.
                // The original exception can be 
                // accessed through the InnerException
                // property.
                if(e.InnerException != null)
                    Console.WriteLine("Success! We caught an exception: " + e.InnerException.Message);
            }

        }
    }
}
