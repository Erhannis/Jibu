Manager Example
Copy Code
using System;
using Jibu;

namespace ManagerExample
{
    // A brief example that explains the
    // usage of the static Manager class.
    class ManagerExample
    {
        static void Main(string[] args)
        {
            // ProcessorCount tells us how many
            // processors/cores that are available
            // to Jibu.
            Console.WriteLine("There are " + Manager.ProcessorCount + " processors available to Jibu.");

            // Initialize Jibu with a stack size
            // of 128 KB. This means that all 
            // threads created by Jibu will have 
            // a stack size of 128 KB.
            // Initialize should be called only
            // once in your program, and before
            // any tasks are started.
            Manager.Initialize(128 * 1024);

            // We can use the StackSize property
            // to get the current stack size.
            Console.WriteLine("Jibu uses a stack size of " + Manager.StackSize + " Bytes.");

            // Run some tasks in parallel
            Parallel.Run(new DelegateAsync (delegate { Console.WriteLine("Hello world!"); }),
                         new DelegateAsync (delegate { Console.WriteLine("Hi mom!"); }),
                         new DelegateAsync (delegate { Console.WriteLine("G'day mate!"); }));
                       
            // To terminate the unused threads in Jibu's 
            // thread pool, you can call the method
            // TerminateFreeThreads. This will terminate 
            // the threads created in Parallel.Run above.
            Manager.TerminateFreeThreads();

        }
    }
}
