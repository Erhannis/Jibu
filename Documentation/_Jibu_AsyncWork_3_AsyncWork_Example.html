using System;
using System.Collections.Generic;
using System.Text;
using System.Threading;
using Jibu;

namespace DelegateExample
{
    class DelegateExample
    {
        static void Main(string[] args)
        {
            // We create a delegate method using 
            // an anonymous class. Unlike when you 
            // subclass Async, it can be called 
            // multiple times. Just remember to 
            // synchronize data-access.

            int i = 0;

            AsyncWork someWork = delegate
            {
                Console.WriteLine("I have been called " + Interlocked.Increment(ref i) + " time(s)");
            };

            // Run someWork 4 times in parallel
            Parallel.Run(new DelegateAsync(someWork),
                    new DelegateAsync(someWork),
                    new DelegateAsync(someWork),
                    new DelegateAsync(someWork));


            i = 0;

            // The FutureWork delegate can be used
            // with a DelegateFuture.
            FutureWork<string> someMoreWork = delegate
            {
                return "I was invoked as number " + Interlocked.Increment(ref i);
            };

            // Run someMoreWork 4 times in parallel 
            // and gather the results.
            // We get the results in the order
            // that we submit them.
            string[] results = Parallel.Gather(new DelegateFuture<string>(someMoreWork),
                    new DelegateFuture<string>(someMoreWork),
                    new DelegateFuture<string>(someMoreWork),
                    new DelegateFuture<string>(someMoreWork));

            foreach (string str in results)
            {
                Console.WriteLine(str);
            }


            // The AsyncWork and FutureWork delegates
            // can acces the actual executing Task
            // through argument given to the method.
            // In the examples above the argument was left
            // out, because it wasn't needed.
            AsyncWork cancelChecker = delegate(Async async)
            {
                // we spin until we're cancelled.
                while (!async.IsCancelled)
                    ;

                Console.WriteLine("I was cancelled!");
            };


            Async a = new DelegateAsync(cancelChecker).Start();

            Thread.Sleep(500);

            a.Cancel();
        }
    }

}
