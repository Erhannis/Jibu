using System;
using System.Collections.Generic;
using System.Text;
using Jibu;

namespace ReduceExample
{
    // This example shows how to combine multiple
    // associative values into a single value, using
    // the Reduce method.
    class ReduceExample
    {
        static void Main(string[] args)
        {
            // Adding integers from 0 to 1000000.
            // loopbody simply returns the index while combineMethod add the numbers.
            int sum1 = Parallel.Reduce<int>(0, 1000000, 0, delegate(int i) { return i; }, delegate(int x, int y) { return x + y; });
            Console.WriteLine("1 + 2 + 3 + ... + 1000000 = " + sum1);
            

            // Adding integers from -10 to 1000000.
            // The initial value is 0 and the grainSize is set to 10000.
            // loopbody simply returns the index while combineMethod add the numbers.
            int sum2 = Parallel.Reduce<int>(-10, 1000000, 0, 10000, delegate(int i) { return i; }, delegate(int x, int y) { return x + y; });
            Console.WriteLine("nn-10 + (-9) + (-8) + ... + 1000000 = " + sum2);
        }
    }
}
